```typescript
interface GitHubReviewRequest {
  body: string;           // Review summary
  event: "APPROVE" | "REQUEST_CHANGES" | "COMMENT";
  comments: Array<{
    path: string;         // File path relative to repo root
    line: number;         // File line number (not diff position)
    side: "LEFT" | "RIGHT";
    body: string;         // Comment text (markdown)
  }>;
}

// Returns: diff between two commits
```

       ```typescript
       export interface ReviewComment {
         path: string;          // file path relative to repo root
         line: number;          // file line number
         side: "LEFT" | "RIGHT"; // LEFT = deletion line, RIGHT = addition/context
         body: string;          // comment text (markdown)
         severity: "critical" | "suggestion" | "nitpick";
       }

       export interface ReviewResult {
         event: "APPROVE" | "REQUEST_CHANGES" | "COMMENT";
         summary: string;
         comments: ReviewComment[];
       }

       export interface ReviewContext {
         prUrl: string;
         prDescription: string;
         existingReviews: string;     // previous review comments for context
         previousReviewSummary?: string; // agent's previous review (for incremental)
         diffSinceLastReview?: string;   // incremental diff
       }

       export interface ReviewProvider {
         fetchDiff(prUrl: string): Promise<string>;
         fetchExistingReviews(prUrl: string): Promise<string>;
         submitReview(prUrl: string, review: ReviewResult): Promise<void>;
         compareDiff(prUrl: string, baseSha: string, headSha: string): Promise<string>;
       }
       ```

    2. Create `server/src/services/review-providers/github.ts`:
       - Export `parsePrUrl(url: string): { owner: string; repo: string; pullNumber: number }` -- extracts from GitHub PR URL
       - Export `buildReviewPayload(review: ReviewResult): GitHubReviewRequest` -- maps ReviewResult to GitHub API format, using `line` + `side` fields (not the deprecated `position` field per Pitfall 4 in research)
       - Export `createGitHubReviewProvider(githubFetchFn: typeof fetch): ReviewProvider`:
         - `fetchDiff(prUrl)`: GET /repos/{owner}/{repo}/pulls/{pull_number} with Accept: application/vnd.github.diff header. Returns raw unified diff string.
         - `fetchExistingReviews(prUrl)`: GET /repos/{owner}/{repo}/pulls/{pull_number}/reviews + /comments. Returns formatted string of existing review comments (author, body, path, line).
         - `submitReview(prUrl, review)`: POST /repos/{owner}/{repo}/pulls/{pull_number}/reviews with mapped payload. Auto-posts directly (no approval gate).
         - `compareDiff(prUrl, baseSha, headSha)`: GET /repos/{owner}/{repo}/compare/{baseSha}...{headSha} with Accept: application/vnd.github.diff. For incremental re-reviews.
       - The `githubFetchFn` parameter is the authenticated `githubFetch` from github-app.ts (uses installation token).
