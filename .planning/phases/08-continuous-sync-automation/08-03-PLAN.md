**1. Add effort classification helper function**

```bash
# Effort estimation: count + type heuristic
classify_effort() {
  local count="$1"
  local has_migration="$2"  # "true" or "false"
  if [ "$count" -ge 6 ]; then
    echo "complex"
  elif [ "$count" -ge 3 ] || [ "$has_migration" = "true" ]; then
    echo "moderate"
  else
    echo "simple"
  fi
}
```
